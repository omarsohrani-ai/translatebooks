// ── PDF Generation (RTL-aware) ───────────────────────────────────────────── // Replace the entire downloadPDF function in your index.html with this. // Also add to : // // async function downloadPDF() { const { jsPDF } = window.jspdf; const doc = new jsPDF({ orientation: 'portrait', unit: 'mm', format: 'a4' }); const W = 210, H = 297, mL = 22, mR = 22, mT = 28, mB = 22; const cW = W - mL - mR; const tgtLangCode = document.getElementById('targetLang').value; const tgtName = document.getElementById('targetLang').options[document.getElementById('targetLang').selectedIndex].text; const srcName = document.getElementById('sourceLang').options[document.getElementById('sourceLang').selectedIndex].text; // RTL languages that need canvas-based rendering const RTL_LANGS = new Set(['ar', 'he', 'fa', 'ur']); const isRTL = RTL_LANGS.has(tgtLangCode); // ── Title page ── doc.setFillColor(13, 17, 23); doc.rect(0, 0, W, H, 'F'); doc.setDrawColor(227, 179, 65); doc.setLineWidth(0.4); doc.rect(8, 8, W-16, H-16); doc.setTextColor(227, 179, 65); doc.setFont('helvetica', 'bold'); doc.setFontSize(24); doc.text('DocuTranslate', W/2, 60, { align: 'center' }); doc.setFontSize(16); doc.setTextColor(240, 246, 252); const titleLines = doc.splitTextToSize('AI-Powered Document Translation', cW); doc.text(titleLines, W/2, 80, { align: 'center' }); doc.setDrawColor(100, 100, 100); doc.line(30, 95, W-30, 95); doc.setFontSize(11); doc.setTextColor(139, 148, 158); doc.setFont('helvetica', 'normal'); doc.text(`Source Language: ${srcName}`, W/2, 110, { align: 'center' }); doc.text(`Translated to: ${tgtName}`, W/2, 120, { align: 'center' }); doc.text(`Pages translated: ${Object.keys(translations).length}`, W/2, 130, { align: 'center' }); doc.text(`Generated: ${new Date().toLocaleString()}`, W/2, 140, { align: 'center' }); doc.setFontSize(9); doc.setTextColor(80, 88, 96); doc.text('Powered by Google Gemini AI · DocuTranslate', W/2, H-20, { align: 'center' }); doc.addPage(); const sorted = Object.keys(translations).map(Number).sort((a,b)=>a-b); // ── RTL path: render via browser → html2canvas → image in PDF ── if (isRTL) { await document.fonts.ready; for (const pn of sorted) { const text = translations[pn] || ''; const isBlank = !text || text.trim() === '[Blank page]' || text.trim() === ''; // White page background doc.setFillColor(255, 255, 255); doc.rect(0, 0, W, H, 'F'); // Header bar doc.setFillColor(13, 17, 23); doc.rect(0, 0, W, 16, 'F'); doc.setFontSize(8); doc.setFont('helvetica', 'italic'); doc.setTextColor(139, 148, 158); doc.text('DocuTranslate', mL, 10); doc.text(`Page ${pn} · ${tgtName}`, W - mR, 10, { align: 'right' }); if (isBlank) { doc.setFontSize(10); doc.setTextColor(180, 180, 180); doc.setFont('helvetica', 'italic'); doc.text(`— Page ${pn}: Blank —`, W/2, H/2, { align: 'center' }); doc.addPage(); continue; } // Render RTL text off-screen using browser fonts // Content area dimensions at 2× scale for crisp output const SCALE = 2; const PX_PER_MM = 96 / 25.4; const divW = Math.round(cW * PX_PER_MM * SCALE); const divH = Math.round((H - 18 - mB) * PX_PER_MM * SCALE); const div = document.createElement('div'); div.style.cssText = [ `position:fixed`, `top:0`, `left:-99999px`, `width:${divW}px`, `height:${divH}px`, `overflow:hidden`, `background:#ffffff`, `color:#1e1e23`, `font-family:'Amiri','Noto Naskh Arabic','Arial Unicode MS',serif`, `font-size:${Math.round(12 * SCALE)}px`, `line-height:2`, `direction:rtl`, `text-align:justify`, `padding:${Math.round(10 * SCALE)}px ${Math.round(14 * SCALE)}px`, `box-sizing:border-box`, `word-break:break-word`, `white-space:pre-wrap` ].join(';'); div.textContent = text; document.body.appendChild(div); try { const canvas = await html2canvas(div, { scale: 1, useCORS: true, backgroundColor: '#ffffff', logging: false, width: divW, height: divH }); const imgData = canvas.toDataURL('image/jpeg', 0.93); // Place image below the 16mm header bar, respecting bottom margin doc.addImage(imgData, 'JPEG', mL, 18, cW, H - 18 - mB); log(`✓ RTL page ${pn} rendered`, 'ok'); } catch (e) { doc.setFontSize(9); doc.setTextColor(200, 0, 0); doc.text(`[Rendering error on page ${pn}]`, mL, mT + 5); log(`⚠ Canvas error on page ${pn}: ${e.message}`, 'warn'); } finally { document.body.removeChild(div); } doc.addPage(); } // ── LTR path: original text-based rendering ── } else { for (const pn of sorted) { const text = translations[pn] || ''; const isBlank = !text || text.trim() === '[Blank page]' || text.trim() === ''; doc.setFillColor(255, 255, 255); doc.rect(0, 0, W, H, 'F'); doc.setFillColor(13, 17, 23); doc.rect(0, 0, W, 16, 'F'); doc.setFontSize(8); doc.setFont('helvetica', 'italic'); doc.setTextColor(139, 148, 158); const fname = `DocuTranslate_${tgtName}_${new Date().toISOString().slice(0,10)}.pdf`; doc.save(fname); log(`✅ Downloaded: ${fname}`, 'ok'); }
